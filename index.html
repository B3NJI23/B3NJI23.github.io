IF használata:

                
# Ha teljesül a megadott paraméter akkor teljesül a beleírt kód

szam = 12

if szam == 12:
    print('A szám egyenlő 12-vel!')

# Továbbá megnézhetünk több paramétert is, ha az első nem teljesül.

szam = 24
                    
if szam == 12:
    print('A szám egyenlő 12-vel!')
elif szam == 24:
    print('A szám egyenlő 24-el')

# Ha egyik feltétel sem teljesül akkor az 'else' fog érvénybe lépni, aminek a használata a következő:

szam = 36

if szam == 12:
    print('A szám egyenlő 12-vel!')
elif szam == 24:
    print('A szám egyenlő 24-el')
else:
    print('A szám nem egyenlő sem 12-vel, sem 24-el')

#Class fogalma: A class egy object (tárgy), amely képes eltárolni különböző adatokat.

    def __init__(self, [saját változók vesszővel elválasztva]):
